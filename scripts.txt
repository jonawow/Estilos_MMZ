    document.addEventListener("DOMContentLoaded", function () {

        function moverBloque() {
            const contenedorDestino = document.querySelector(".avance-marker");
            const bloqueAvance = document.querySelector(".block_completion_progress, .block_dedication");

            if (!contenedorDestino || !bloqueAvance) return false;

            // evitar moverlo varias veces
            if (contenedorDestino.contains(bloqueAvance)) return true;

            contenedorDestino.parentNode.insertBefore(bloqueAvance, contenedorDestino.nextSibling);

            // estilos opcionales
            bloqueAvance.style.marginBottom = "0";
            bloqueAvance.style.width = "100%";

            return true;
        }

        // intento inmediato
        if (moverBloque()) return;

        // por one_topic: esperar cambios en el DOM
        const observer = new MutationObserver(() => {
            if (moverBloque()) {
                observer.disconnect();
            }
        });

        observer.observe(document.body, { childList: true, subtree: true });

    });

    document.addEventListener("DOMContentLoaded", function () {
        // 1. Obtener ID real del curso desde el body (nunca falla, ni en PDFs)
        var matchCurso = document.body.className.match(/course-(\d+)/);
        var idCurso = matchCurso ? matchCurso[1] : null;

        // 2. Candado de seguridad: Evitar que salga en la portada general del sitio
        if (!idCurso || idCurso === '1' || document.body.classList.contains('path-site') || document.body.classList.contains('path-my')) {
            return;
        }

        // Caja fuerte en memoria exclusiva para este curso
        var keyMemoriaChico = 'banner_chico_curso_' + idCurso;

        // Buscamos los bloques f├¡sicos en la p├ígina
        var bloqueGrandeFisico = document.querySelector('.block_html .banner-grande');
        var bloqueChicoFisico = document.querySelector('.block_html .banner-chico, .block_html .banner-modulo:not(.banner-grande)');
        var bloqueCapacitacionFisico = document.querySelector('.block_html .banner-capacitacion');
        var bannerParaClonar = null;

        // A. ┬┐ESTAMOS EN LA PORTADA DEL CURSO? (Vemos los bloques f├¡sicos)
        if (bloqueGrandeFisico || bloqueChicoFisico || bloqueCapacitacionFisico) {

            if (bloqueCapacitacionFisico) {
                bannerParaClonar = bloqueCapacitacionFisico;
            } else if (bloqueGrandeFisico) {
                bannerParaClonar = bloqueGrandeFisico; // Elegimos el grande para mostrar en la portada
            }

            if (bloqueChicoFisico) {
                // Guardamos el CHICO en la memoria en secreto para usarlo en los recursos
                sessionStorage.setItem(keyMemoriaChico, bloqueChicoFisico.outerHTML);
            }

            // Escondemos TODOS los bloques del panel lateral (a menos que el profe est├® editando)
            if (!document.body.classList.contains('editing')) {
                document.querySelectorAll('.block_html').forEach(function (bloque) {
                    if (bloque.querySelector('.banner-grande') || bloque.querySelector('.banner-chico') || bloque.querySelector('.banner-capacitacion') || bloque.querySelector('.banner-modulo')) {
                        bloque.style.display = 'none';
                    }
                });
            }
        }
        // B. Banners chicos para recurso, Moodle escondi├│ los paneles
        else {
            // Sacamos el banner CHICO de la memoria
            var bannerGuardado = sessionStorage.getItem(keyMemoriaChico);
            if (bannerGuardado) {
                var tempDiv = document.createElement('div');
                tempDiv.innerHTML = bannerGuardado;
                bannerParaClonar = tempDiv.firstElementChild;
            }

            var areaContenido = document.querySelector('#region-main') || document.querySelector('[role="main"]');

                var bannerFinal = bannerParaClonar.cloneNode(true);
                bannerFinal.classList.add('banner-magico');
                areaContenido.prepend(bannerFinal);
            }
        }
    });

    document.addEventListener('DOMContentLoaded', function () {
        // Variable para detectar si es estudiante (sin bot├│n de edici├│n)
        var esEstudiante = !document.querySelector('[data-action="toggle-editmode"], .editmode-switch-form');
        var fechaBase = null;
        var etiquetaMaestra = null;

        // INTENTO 1: Buscar en Smart Label generada autom├íticamente (ej. "Inicio del curso: 7/04/26")
        var xpathSmart = "//text()[contains(., 'Inicio del curso:')]";
        var resultSmart = document.evaluate(xpathSmart, document.body, null, XPathResult.FIRST_ORDERED_NODE_TYPE, null);

        if (resultSmart.singleNodeValue) {
            var textoNodo = resultSmart.singleNodeValue;
            var contenedorPadre = textoNodo.parentElement;
            etiquetaMaestra = contenedorPadre.closest('li.activity') || contenedorPadre.closest('.activity-item');

            var codeElement = contenedorPadre.querySelector('code');
            var dateText = null;

            if (codeElement) {
                dateText = codeElement.innerText.trim();
            } else {
                var matchRegex = contenedorPadre.innerText.match(/Inicio del curso:\s*(\d{1,2}\/\d{1,2}\/\d{2,4})/i);
                if (matchRegex) dateText = matchRegex[1];
            }

            if (dateText) {
                // Formato DD/MM/YY o DD/MM/YYYY
                var partes = dateText.split('/');
                if (partes.length === 3) {
                    var dia = parseInt(partes[0], 10);
                    var mes = parseInt(partes[1], 10) - 1;
                    var ano = parseInt(partes[2], 10);
                    if (ano < 100) ano += 2000;
                    fechaBase = new Date(ano, mes, dia);
                }
            }
        }

        // INTENTO 2: Buscar el tag manual antiguo {FECHA_INICIO: YYYY-MM-DD} por si acaso
        if (!fechaBase) {
            var xpathManual = "//text()[contains(., '{FECHA_INICIO:')]";
            var resultManual = document.evaluate(xpathManual, document.body, null, XPathResult.FIRST_ORDERED_NODE_TYPE, null);

            if (resultManual.singleNodeValue) {
                var textNodeManual = resultManual.singleNodeValue;
                etiquetaMaestra = textNodeManual.parentElement.closest('li.activity') || textNodeManual.parentElement.closest('.activity-item');
                var match = textNodeManual.nodeValue.match(/\{FECHA_INICIO:\s*(\d{4}-\d{2}-\d{2})\}/);

                if (match) {
                    // Ocultar texto original
                    if (textNodeManual.parentElement && textNodeManual.parentElement.tagName === 'P') {
                        textNodeManual.parentElement.style.display = 'none';
                    }
                    var partes = match[1].split('-');
                    if (partes.length === 3) {
                        fechaBase = new Date(partes[0], partes[1] - 1, partes[2]);
                    }
                }
            }
        }

        // SI ENCONTRAMOS UNA FECHA DE INICIO V├üLIDA
        if (fechaBase) {

            // 1. El c├│digo original ocultaba toda la etiqueta al estudiante. Lo hemos retirado
            // para que los datos del plugin Smart Label s├¡ se muestren.

            var meses = ['Enero', 'Febrero', 'Marzo', 'Abril', 'Mayo', 'Junio', 'Julio', 'Agosto', 'Septiembre', 'Octubre', 'Noviembre', 'Diciembre'];

            // Funci├│n matem├ítica para sumar semanas
            function obtenerRangoFechas(semanasSumadas) {
                var inicio = new Date(fechaBase);
                inicio.setDate(fechaBase.getDate() + (semanasSumadas * 7));
                var fin = new Date(inicio);
                fin.setDate(inicio.getDate() + 6);
                return inicio.getDate() + ' ' + meses[inicio.getMonth()] + ' - ' + fin.getDate() + ' ' + meses[fin.getMonth()];
            }

            // 2. Buscamos tus identificadores
            var walker = document.createTreeWalker(document.body, NodeFilter.SHOW_TEXT, null, false);
            var nodo;
            var nodosConMarcador = [];

            while (nodo = walker.nextNode()) {
                if (nodo.nodeValue.includes('{RANGO_')) {
                    nodosConMarcador.push(nodo);
                }
            }

            // 3. Reemplazamos creando la estructura HTML para el CSS
            nodosConMarcador.forEach(function (nodoTexto) {
                var texto = nodoTexto.nodeValue;
                var nuevoHTML = "";

                for (var i = 1; i <= 10; i++) {
                    if (texto.includes('{RANGO_S' + i + '}')) {
                        nuevoHTML = '<span class="rango-semana-dinamico"><span class="rango-semana-badge">S' + i + '</span><span class="rango-semana-sep">|</span><span class="rango-semana-fechas">' + obtenerRangoFechas(i - 1) + '</span></span>';
                    }
                }

                if (texto.includes('{RANGO_RECESO}')) {
                    nuevoHTML = '<span class="rango-semana-dinamico"><span class="rango-semana-badge">Receso</span><span class="rango-semana-sep">|</span><span class="rango-semana-fechas">' + obtenerRangoFechas(6) + '</span></span>';
                }

                // Reemplazamos el texto por nuestro nuevo HTML
                if (nuevoHTML !== "") {
                    var spanContenedor = document.createElement('span');
                    spanContenedor.innerHTML = nuevoHTML;
                    nodoTexto.parentNode.replaceChild(spanContenedor, nodoTexto);
                }
            });
        }
    });
</script>

    function setupNavArrow(arrowSelector, linkSelector) {
        const arrow = document.querySelector(arrowSelector);
        const link = document.querySelector(linkSelector);

        if (!arrow || !link) return;

        // Evitar registrar listeners duplicados
        if (arrow.dataset.navBound) return;
        arrow.dataset.navBound = "1";

        // Atributos de accesibilidad
        arrow.setAttribute("role", "button");
        arrow.setAttribute("tabindex", "0");
        arrow.setAttribute("aria-label", link.getAttribute("aria-label") ?? link.textContent.trim());

        const navigate = () => { window.location.href = link.href; };

        // Click + Enter/Space para teclado
        arrow.addEventListener("click", navigate);
        arrow.addEventListener("keydown", (e) => {
            if (e.key === "Enter" || e.key === " ") {
                e.preventDefault();
                navigate();
            }
        });
    }
</script>

    // -----------------------------------------------------------------------
    //  Renombrar tooltips del drawer de bloques
    //  "Caj├│n del bloque" ÔåÆ "Cerrar/Abrir panel de apoyo"
    //  Se desconecta el observer antes de modificar atributos para evitar
    //  el bucle infinito: cambio ÔåÆ disparo ÔåÆ cambio ÔåÆ disparo...
    // -----------------------------------------------------------------------
    (function () {
        var LABEL_CERRAR = 'Cerrar panel de apoyo';
        var LABEL_ABRIR = 'Abrir panel de apoyo';
        var SELECTOR_CERRAR = '[data-action="closedrawer"][data-target="theme_boost-drawers-blocks"]';
        var SELECTOR_ABRIR = '[data-action="toggle"][data-target="theme_boost-drawers-blocks"]';

        function actualizarLabel(el, texto) {
            if (!el) return;
            // Solo tocar si el valor es diferente (evita mutaciones innecesarias)
            if (el.getAttribute('aria-label') !== texto) el.setAttribute('aria-label', texto);
            if (el.getAttribute('data-bs-original-title') !== texto) el.setAttribute('data-bs-original-title', texto);
            // NO tocar 'title': lo usa el navegador para el tooltip cuadrado nativo,
            // que se duplicar├¡a con el tooltip de Bootstrap ya inicializado.
            el.removeAttribute('title'); // limpiar si qued├│ de antes
            var sr = el.querySelector('.sr-only');
            if (sr && sr.textContent !== texto) sr.textContent = texto;
        }

        function renombrarTooltips() {
            // Pausar el observer ANTES de modificar atributos
            observer.disconnect();

            document.querySelectorAll(SELECTOR_CERRAR).forEach(function (btn) {
                actualizarLabel(btn, LABEL_CERRAR);
            });

            document.querySelectorAll(SELECTOR_ABRIR).forEach(function (btn) {
                actualizarLabel(btn, LABEL_ABRIR);

                // El bot├│n "Abrir" no tiene tooltip Bootstrap inicializado (usa data-disabled-toggle).
                // Lo inicializamos manualmente para que use el mismo estilo redondeado.
                btn.setAttribute('data-bs-toggle', 'tooltip');
                btn.setAttribute('data-bs-placement', 'right');
                if (typeof bootstrap !== 'undefined' && bootstrap.Tooltip) {
                    var tip = bootstrap.Tooltip.getInstance(btn);
                    if (!tip) {
                        new bootstrap.Tooltip(btn);
                    } else {
                        tip.setContent({ '.tooltip-inner': LABEL_ABRIR });
                    }
                }
            });

            // Reconectar despu├®s de terminar los cambios
            observer.observe(document.documentElement, OBSERVER_CONFIG);
        }

        var OBSERVER_CONFIG = {
            childList: true,
            subtree: true,
            attributes: true,
            attributeFilter: ['data-bs-original-title', 'aria-label', 'title']
        };

        var observer = new MutationObserver(function (mutations) {
            var hayNodosNuevos = mutations.some(function (m) { return m.addedNodes.length > 0; });
            var hayAtributosCajon = mutations.some(function (m) {
                return m.type === 'attributes' &&
                    m.target.matches &&
                    (m.target.matches(SELECTOR_CERRAR) || m.target.matches(SELECTOR_ABRIR));
            });

            if (hayNodosNuevos || hayAtributosCajon) renombrarTooltips();
        });

        // Ejecutar al cargar el DOM
        if (document.readyState === 'loading') {
            document.addEventListener('DOMContentLoaded', function () {
                renombrarTooltips();
            });
        } else {
            renombrarTooltips();
        }
    })();
</script>

    document.addEventListener("DOMContentLoaded", function () {
        // Solo actuar si estamos en la vista de un recurso
        if (!document.body.classList.contains('path-mod-resource')) return;

        // Buscar si hay contenido de recurso
        var resourceContent = document.querySelector('.resourcecontent');
        if (!resourceContent) return;

        // Verificar si es un PDF (busca iframe o object o embed que cargue PDF)
        var pdfEmbed = resourceContent.querySelector('iframe, object, embed');
        if (!pdfEmbed) return;

        var pdfUrl = pdfEmbed.src || pdfEmbed.data;
        if (!pdfUrl) return;

        // Verificar que realmente sea un PDF (por extensi├│n o por tipo de objeto)
        var isPdf = pdfUrl.toLowerCase().includes('.pdf') || (pdfEmbed.tagName.toLowerCase() === 'object' && pdfEmbed.getAttribute('type') === 'application/pdf');
        if (!isPdf) return;

        // Crear el contenedor de los botones
        var btnContainer = document.createElement('div');
        btnContainer.className = 'pdf-buttons-container';

        // Preparar URL de descarga (asegurar que fuerce descarga si es Moodle pluginfile)
        var downloadUrl = pdfUrl;
        if (downloadUrl.includes('pluginfile.php') && !downloadUrl.includes('forcedownload=1')) {
            downloadUrl += (downloadUrl.includes('?') ? '&' : '?') + 'forcedownload=1';
        }

        // Bot├│n Descargar
        var btnDownload = document.createElement('a');
        btnDownload.href = downloadUrl;
        btnDownload.className = 'btn-custom-pdf';
        btnDownload.innerText = 'Descargar PDF';
        btnDownload.setAttribute('download', '');

        // Bot├│n Pantalla Completa
        var btnFullscreen = document.createElement('button');
        btnFullscreen.className = 'btn-custom-pdf';
        btnFullscreen.innerText = 'Pantalla completa';
        btnFullscreen.onclick = function () {
            window.open(pdfUrl, '_blank');
        };

        btnContainer.appendChild(btnFullscreen);

    });
